CREATE TABLE COMPANIES(
   COMPANYID 	INT PRIMARY KEY		NOT NULL,
   NAME			varchar(60)    		NOT NULL,
   INDUSTRY		varchar(60)     	NOT NULL,
   CREATED		TIMESTAMP,
   UPDATED		TIMESTAMP
);
CREATE TABLE INTERVIEWERS(
   INTERVIEWERID 		INT PRIMARY KEY		NOT NULL,
   INTERVIEWER			varchar(60)    		NOT NULL,
   POSITION				varchar(60)    		NOT NULL,
   COMPANYID			INT 				NOT NULL
);
CREATE TABLE QUESTIONS(
   QUESTIONID	INT PRIMARY KEY			NOT NULL,
   QUESTION 	TEXT    				NOT NULL,
   JOBID		INT     				NOT NULL,
   TOPICID		INT 					NOT NULL,
   CREATED		TIMESTAMP,
   UPDATED		TIMESTAMP
);
CREATE TABLE ANSWERS(
   ANSWERID 	INT PRIMARY KEY     NOT NULL,
   ANSWER 		TEXT    			NOT NULL,
   QUESTIONID   INT     			NOT NULL,
   CREATED		TIMESTAMP,
   UPDATED		TIMESTAMP
);
CREATE TABLE JOBSTYPES(
   JOBID		INT PRIMARY KEY			NOT NULL,
   JOBPOSITION	varchar(60)				NOT NULL	
);
CREATE TABLE TOPICS(
   TOPICID		INT PRIMARY KEY			NOT NULL,
   TOPICNAME	varchar(60)				NOT NULL	
);
