   COMPANYID 	SERIAL PRIMARY KEY		NOT NULL,
   NAME			varchar(60)    		NOT NULL,
   INDUSTRY		varchar(60)     	NOT NULL,
   CREATED		TIMESTAMP 			DEFAULT NOW(),
   UPDATED		TIMESTAMP			DEFAULT NOW()
